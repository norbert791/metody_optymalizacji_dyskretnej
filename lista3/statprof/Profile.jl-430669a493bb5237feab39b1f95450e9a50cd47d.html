34 # triggers printing the report and (optionally) saving a heap snapshot after a SIGINFO/SIGUSR1 profile request
35 const PROFILE_PRINT_COND = Ref{Base.AsyncCondition}()
36 function profile_printing_listener()
37 try
38 while true
39 wait(PROFILE_PRINT_COND[])
40 peek_report[]()
41 if get(ENV, "JULIA_PROFILE_PEEK_HEAP_SNAPSHOT", nothing) === "1"
42 println(stderr, "Saving heap snapshot...")
43 fname = take_heap_snapshot()
44 println(stderr, "Heap snapshot saved to `$(fname)`")
45 end
46 end
47 catch ex
48 if !isa(ex, InterruptException)
49 @error "Profile printing listener crashed" exception=ex,catch_backtrace()
50 end
51 end
52 end

88 """
89 init(; n::Integer, delay::Real)
90
91 Configure the `delay` between backtraces (measured in seconds), and the number `n` of instruction pointers that may be
92 stored per thread. Each instruction pointer corresponds to a single line of code; backtraces generally consist of a long
93 list of instruction pointers. Note that 6 spaces for instruction pointers per backtrace are used to store metadata and two
94 NULL end markers. Current settings can be obtained by calling this function with no arguments, and each can be set independently
95 using keywords or in the order `(n, delay)`.
96 """
97 function init(; n::Union{Nothing,Integer} = nothing, delay::Union{Nothing,Real} = nothing, limitwarn::Bool = true)
98 n_cur = ccall(:jl_profile_maxlen_data, Csize_t, ())
99 if n_cur == 0 && isnothing(n) && isnothing(delay)
100 # indicates that the buffer hasn't been initialized at all, so set the default
101 default_init()
102 n_cur = ccall(:jl_profile_maxlen_data, Csize_t, ())
103 end
104 delay_cur = ccall(:jl_profile_delay_nsec, UInt64, ())/10^9
105 if n === nothing && delay === nothing
106 return n_cur, delay_cur
107 end
108 nnew = (n === nothing) ? n_cur : n
109 delaynew = (delay === nothing) ? delay_cur : delay
110 init(nnew, delaynew; limitwarn)
111 end

113 function init(n::Integer, delay::Real; limitwarn::Bool = true)
114 sample_size_bytes = sizeof(Ptr) # == Sys.WORD_SIZE / 8
115 buffer_samples = n
116 buffer_size_bytes = buffer_samples * sample_size_bytes
117 if buffer_size_bytes > 2^29 && Sys.WORD_SIZE == 32
118 buffer_samples = floor(Int, 2^29 / sample_size_bytes)
119 buffer_size_bytes = buffer_samples * sample_size_bytes
120 limitwarn && @warn "Requested profile buffer limited to 512MB (n = $buffer_samples) given that this system is 32-bit"
121 end
122 status = ccall(:jl_profile_init, Cint, (Csize_t, UInt64), buffer_samples, round(UInt64, 10^9*delay))
123 if status == -1
124 error("could not allocate space for ", n, " instruction pointers ($(Base.format_bytes(buffer_size_bytes)))")
125 end
126 end

128 function default_init()
129 # init with default values
130 # Use a max size of 10M profile samples, and fire timer every 1ms
131 # (that should typically give around 100 seconds of record)
132 @static if Sys.iswindows() && Sys.WORD_SIZE == 32
133 # The Win32 unwinder is 1000x slower than elsewhere (around 1ms/frame),
134 # so we don't want to slow the program down by quite that much
135 n = 1_000_000
136 delay = 0.01
137 else
138 # Keep these values synchronized with trigger_profile_peek
139 n = 10_000_000
140 delay = 0.001
141 end
142 init(n, delay, limitwarn = false)
143 end

201 """
202 print([io::IO = stdout,] [data::Vector = fetch()], [lidict::Union{LineInfoDict, LineInfoFlatDict} = getdict(data)]; kwargs...)
203
204 Prints profiling results to `io` (by default, `stdout`). If you do not
205 supply a `data` vector, the internal buffer of accumulated backtraces
206 will be used.
207
208 The keyword arguments can be any combination of:
209
210 - `format` -- Determines whether backtraces are printed with (default, `:tree`) or without (`:flat`)
211 indentation indicating tree structure.
212
213 - `C` -- If `true`, backtraces from C and Fortran code are shown (normally they are excluded).
214
215 - `combine` -- If `true` (default), instruction pointers are merged that correspond to the same line of code.
216
217 - `maxdepth` -- Limits the depth higher than `maxdepth` in the `:tree` format.
218
219 - `sortedby` -- Controls the order in `:flat` format. `:filefuncline` (default) sorts by the source
220 line, `:count` sorts in order of number of collected samples, and `:overhead` sorts by the number of samples
221 incurred by each function by itself.
222
223 - `groupby` -- Controls grouping over tasks and threads, or no grouping. Options are `:none` (default), `:thread`, `:task`,
224 `[:thread, :task]`, or `[:task, :thread]` where the last two provide nested grouping.
225
226 - `noisefloor` -- Limits frames that exceed the heuristic noise floor of the sample (only applies to format `:tree`).
227 A suggested value to try for this is 2.0 (the default is 0). This parameter hides samples for which `n <= noisefloor * N`,
228 where `n` is the number of samples on this line, and `N` is the number of samples for the callee.
229
230 - `mincount` -- Limits the printout to only those lines with at least `mincount` occurrences.
231
232 - `recur` -- Controls the recursion handling in `:tree` format. `:off` (default) prints the tree as normal. `:flat` instead
233 compresses any recursion (by ip), showing the approximate effect of converting any self-recursion into an iterator.
234 `:flatc` does the same but also includes collapsing of C frames (may do odd things around `jl_apply`).
235
236 - `threads::Union{Int,AbstractVector{Int}}` -- Specify which threads to include snapshots from in the report. Note that
237 this does not control which threads samples are collected on (which may also have been collected on another machine).
238
239 - `tasks::Union{Int,AbstractVector{Int}}` -- Specify which tasks to include snapshots from in the report. Note that this
240 does not control which tasks samples are collected within.
241 """

288 taskids = intersect(get_task_ids(data, threadid), tasks)
289 if length(taskids) == 0
290 any_nosamples = true
291 else
292 nl = length(taskids) > 1 ? "\n" : ""
293 printstyled(io, "Thread $threadid$nl"; bold=true, color=Base.info_color())
294 for taskid in taskids
295 printstyled(io, " Task $(Base.repr(taskid)) "; bold=true, color=Base.debug_color())
296 nosamples = print(io, data, lidict, pf, format, threadid, taskid, true)

298 println(io)

301 end
302 elseif groupby === :task

355 taskids = UInt[]

380 function is_block_end(data, i)
381 i < nmeta + 1 && return false
382 # 32-bit linux has been seen to have rogue NULL ips, so we use two to
383 # indicate block end, where the 2nd is the actual end index.
384 # and we could have (though very unlikely):
385 # 1:<stack><metadata><null><null><NULL><metadata><null><null>:end
386 # and we want to ignore the triple NULL (which is an ip).
387 return data[i] == 0 && data[i - 1] == 0 && data[i - META_OFFSET_SLEEPSTATE] != 0
388 end
389
390 function has_meta(data)
391 for i in 6:length(data)
392 data[i] == 0 || continue # first block end null
393 data[i - 1] == 0 || continue # second block end null
394 data[i - META_OFFSET_SLEEPSTATE] in 1:2 || continue
395 data[i - META_OFFSET_CPUCYCLECLOCK] != 0 || continue
396 data[i - META_OFFSET_TASKID] != 0 || continue
397 data[i - META_OFFSET_THREADID] != 0 || continue
398 return true
399 end
400 return false
401 end

421 function getdict!(dict::LineInfoDict, data::Vector{UInt})
422 # we don't want metadata here as we're just looking up ips
423 unique_ips = unique(has_meta(data) ? strip_meta(data) : data)
424 n_unique_ips = length(unique_ips)
425 n_unique_ips == 0 && return dict
426 iplookups = similar(unique_ips, Vector{StackFrame})
427 @sync for indexes_part in Iterators.partition(eachindex(unique_ips), div(n_unique_ips, Threads.threadpoolsize(), RoundUp))
428 Threads.@spawn begin
429 for i in indexes_part
430 iplookups[i] = _lookup_corrected(unique_ips[i])
431 end
432 end
433 end
434 for i in eachindex(unique_ips)
435 dict[unique_ips[i]] = iplookups[i]
436 end
437 return dict
438 end

440 function _lookup_corrected(ip::UInt)
441 st = lookup(convert(Ptr{Cvoid}, ip))
442 # To correct line numbers for moving code, put it in the form expected by

450 """
451 flatten(btdata::Vector, lidict::LineInfoDict) -> (newdata::Vector{UInt64}, newdict::LineInfoFlatDict)
452
453 Produces "flattened" backtrace data. Individual instruction pointers
454 sometimes correspond to a multi-frame backtrace due to inlining; in
455 such cases, this function inserts fake instruction pointers for the
456 inlined calls, and returns a dictionary that is a 1-to-1 mapping
457 between instruction pointers and a single StackFrame.
458 """
459 function flatten(data::Vector, lidict::LineInfoDict)
460 # Makes fake instruction pointers, counting down from typemax(UInt)
461 newip = typemax(UInt64) - 1
462 taken = Set(keys(lidict)) # make sure we don't pick one that's already used
463 newdict = Dict{UInt64,StackFrame}()
464 newmap = Dict{UInt64,Vector{UInt64}}()
465 for (ip, trace) in lidict
466 if length(trace) == 1
467 newdict[ip] = trace[1]
468 else
469 newm = UInt64[]
470 for sf in trace
471 while newip taken && newip > 0
472 newip -= 1
473 end
474 newip == 0 && error("all possible instruction pointers used")
475 push!(newm, newip)
476 newdict[newip] = sf
477 newip -= 1
478 end
479 newmap[ip] = newm
480 end
481 end
482 newdata = UInt64[]
483 for ip::UInt64 in data
484 if haskey(newmap, ip)
485 append!(newdata, newmap[ip])
486 else
487 push!(newdata, ip)
488 end
489 end
490 return (newdata, newdict)
491 end

493 # Take a file-system path and try to form a concise representation of it
494 # based on the package ecosystem
495 function short_path(spath::Symbol, filenamecache::Dict{Symbol, String})
496 return get!(filenamecache, spath) do
497 path = string(spath)
498 if isabspath(path)
499 if ispath(path)
500 # try to replace the file-system prefix with a short "@Module" one,
501 # assuming that profile came from the current machine
502 # (or at least has the same file-system layout)
503 root = path
504 while !isempty(root)
505 root, base = splitdir(root)
506 isempty(base) && break
507 @assert startswith(path, root)
508 for proj in Base.project_names
509 project_file = joinpath(root, proj)
510 if Base.isfile_casesensitive(project_file)
511 pkgid = Base.project_file_name_uuid(project_file, "")
512 isempty(pkgid.name) && return path # bad Project file
513 # return the joined the module name prefix and path suffix
514 path = path[nextind(path, sizeof(root)):end]
515 return string("@", pkgid.name, path)
516 end
517 end
518 end
519 end
520 return path
521 elseif isfile(joinpath(Sys.BINDIR, Base.DATAROOTDIR, "julia", "base", path))
522 # do the same mechanic for Base (or Core/Compiler) files as above,
523 # but they start from a relative path
524 return joinpath("@Base", normpath(path))
525 else
526 # for non-existent relative paths (such as "REPL[1]"), just consider simplifying them
527 return normpath(path) # drop leading "./"

613 fetch(;include_meta = true) -> data
614
615 Return a copy of the buffer of profile backtraces. Note that the
616 values in `data` have meaning only on this machine in the current session, because it
617 depends on the exact memory addresses used in JIT-compiling. This function is primarily for
618 internal use; [`retrieve`](@ref) may be a better choice for most users.
619 By default metadata such as threadid and taskid is included. Set `include_meta` to `false` to strip metadata.
620 """
621 function fetch(;include_meta = true, limitwarn = true)
622 maxlen = maxlen_data()
623 if maxlen == 0
624 error("The profiling data buffer is not initialized. A profile has not been requested this session.")
625 end
626 len = len_data()
627 if limitwarn && is_buffer_full()
628 @warn """The profile data buffer is full; profiling probably terminated
629 before your program finished. To profile for longer runs, call
630 `Profile.init()` with a larger buffer and/or larger delay."""
631 end
632 data = Vector{UInt}(undef, len)
633 GC.@preserve data unsafe_copyto!(pointer(data), get_data_pointer(), len)
634 if include_meta || isempty(data)
635 return data
636 end
637 return strip_meta(data)
638 end
639
640 function strip_meta(data)
641 nblocks = count(Base.Fix1(is_block_end, data), eachindex(data))
642 data_stripped = Vector{UInt}(undef, length(data) - (nblocks * (nmeta + 1)))
643 j = length(data_stripped)
644 i = length(data)
645 while i > 0 && j > 0
646 data_stripped[j] = data[i]
647 if is_block_end(data, i)
648 i -= (nmeta + 1) # metadata fields and the extra NULL IP
649 end
650 i -= 1
651 j -= 1
652 end
653 @assert i == j == 0 "metadata stripping failed"
654 return data_stripped
655 end

657 """
658 Profile.add_fake_meta(data; threadid = 1, taskid = 0xf0f0f0f0) -> data_with_meta
659
660 The converse of `Profile.fetch(;include_meta = false)`; this will add fake metadata, and can be used
661 for compatibility and by packages (e.g., FlameGraphs.jl) that would rather not depend on the internal
662 details of the metadata format.
663 """
664 function add_fake_meta(data; threadid = 1, taskid = 0xf0f0f0f0)
665 threadid == 0 && error("Fake threadid cannot be 0")
666 taskid == 0 && error("Fake taskid cannot be 0")
667 !isempty(data) && has_meta(data) && error("input already has metadata")
668 cpu_clock_cycle = UInt64(99)
669 data_with_meta = similar(data, 0)
670 for i = 1:length(data)
671 val = data[i]
672 if iszero(val)
673 # (threadid, taskid, cpu_cycle_clock, thread_sleeping)
674 push!(data_with_meta, threadid, taskid, cpu_clock_cycle+=1, false+1, 0, 0)
675 else
676 push!(data_with_meta, val)
677 end
678 end
679 return data_with_meta
680 end

682 ## Print as a flat list
683 # Counts the number of times each line appears, at any nesting level and at the topmost level
684 # Merging multiple equivalent entries and recursive calls
685 function parse_flat(::Type{T}, data::Vector{UInt64}, lidict::Union{LineInfoDict, LineInfoFlatDict}, C::Bool,
686 threads::Union{Int,AbstractVector{Int}}, tasks::Union{UInt,AbstractVector{UInt}}) where {T}
687 !isempty(data) && !has_meta(data) && error("Profile data is missing required metadata")
688 lilist = StackFrame[]
689 n = Int[]
690 m = Int[]
691 lilist_idx = Dict{T, Int}()
692 recursive = Set{T}()
693 first = true
694 totalshots = 0
695 startframe = length(data)
696 skip = false
697 nsleeping = 0
698 for i in startframe:-1:1
699 (startframe - 1) >= i >= (startframe - (nmeta + 1)) && continue # skip metadata (its read ahead below) and extra block end NULL IP
700 ip = data[i]
701 if is_block_end(data, i)
702 # read metadata
703 thread_sleeping = data[i - META_OFFSET_SLEEPSTATE] - 1 # subtract 1 as state is incremented to avoid being equal to 0
704 # cpu_cycle_clock = data[i - META_OFFSET_CPUCYCLECLOCK]
705 taskid = data[i - META_OFFSET_TASKID]
706 threadid = data[i - META_OFFSET_THREADID]
707 if !in(threadid, threads) || !in(taskid, tasks)
708 skip = true
709 continue
710 end
711 if thread_sleeping == 1
712 nsleeping += 1
713 end
714 skip = false
715 totalshots += 1
716 empty!(recursive)
717 first = true
718 startframe = i
719 elseif !skip
720 frames = lidict[ip]
721 nframes = (frames isa Vector ? length(frames) : 1)
722 for j = 1:nframes
723 frame = (frames isa Vector ? frames[j] : frames)
724 !C && frame.from_c && continue
725 key = (T === UInt64 ? ip : frame)
726 idx = get!(lilist_idx, key, length(lilist) + 1)
727 if idx > length(lilist)
728 push!(recursive, key)
729 push!(lilist, frame)
730 push!(n, 1)
731 push!(m, 0)
732 elseif !(key in recursive)
733 push!(recursive, key)
734 n[idx] += 1
735 end
736 if first
737 m[idx] += 1
738 first = false
739 end
740 end
741 end
742 end
743 @assert length(lilist) == length(n) == length(m) == length(lilist_idx)
744 return (lilist, n, m, totalshots, nsleeping)
745 end

774 end

784 elseif fmt.sortedby === :overhead

847 ## A tree representation
848
849 # Representation of a prefix trie of backtrace counts
850 mutable struct StackFrameTree{T} # where T <: Union{UInt64, StackFrame}
851 # content fields:
852 frame::StackFrame
853 count::Int # number of frames this appeared in
854 overhead::Int # number frames where this was the code being executed
855 flat_count::Int # number of times this frame was in the flattened representation (unlike count, this'll sum to 100% of parent)
856 max_recur::Int # maximum number of times this frame was the *top* of the recursion in the stack
857 count_recur::Int # sum of the number of times this frame was the *top* of the recursion in a stack (divide by count to get an average)
858 down::Dict{T, StackFrameTree{T}}
859 # construction workers:
860 recur::Int
861 builder_key::Vector{UInt64}
862 builder_value::Vector{StackFrameTree{T}}
863 up::StackFrameTree{T}
864 StackFrameTree{T}() where {T} = new(UNKNOWN, 0, 0, 0, 0, 0, Dict{T, StackFrameTree{T}}(), 0, UInt64[], StackFrameTree{T}[])
865 end

868 const indent_s = " "^10
869 const indent_z = collect(eachindex(indent_s))
870 function indent(depth::Int)
871 depth < 1 && return ""
872 depth <= length(indent_z) && return indent_s[1:indent_z[depth]]
873 div, rem = divrem(depth, length(indent_z))
874 indent = indent_s^div
875 rem != 0 && (indent *= SubString(indent_s, 1, indent_z[rem]))
876 return indent
877 end

879 function tree_format(frames::Vector{<:StackFrameTree}, level::Int, cols::Int, maxes, filenamemap::Dict{Symbol,String}, showpointer::Bool)
880 nindent = min(cols>>1, level)
881 ndigoverhead = ndigits(maxes.overhead)
882 ndigcounts = ndigits(maxes.count)
883 ndigline = ndigits(maximum(frame.frame.line for frame in frames)) + 6
884 ntext = max(30, cols - ndigoverhead - nindent - ndigcounts - ndigline - 6)
885 widthfile = 2*ntext5 # min 12
886 widthfunc = 3*ntext5 # min 18
887 strs = Vector{String}(undef, length(frames))
888 showextra = false
889 if level > nindent
890 nextra = level - nindent
891 nindent -= ndigits(nextra) + 2
892 showextra = true
893 end
894 for i = 1:length(frames)
895 frame = frames[i]
896 li = frame.frame
897 stroverhead = lpad(frame.overhead > 0 ? string(frame.overhead) : "", ndigoverhead, " ")
898 base = nindent == 0 ? "" : indent(nindent - 1) * " "
899 if showextra
900 base = string(base, "+", nextra, " ")
901 end
902 strcount = rpad(string(frame.count), ndigcounts, " ")
903 if li != UNKNOWN
904 if li.line == li.pointer
905 strs[i] = string(stroverhead, "", base, strcount, " ",
906 "[unknown function] (pointer: 0x",
907 string(li.pointer, base = 16, pad = 2*sizeof(Ptr{Cvoid})),
908 ")")
909 else
910 if !li.from_c && li.linfo !== nothing
911 fname = sprint(show_spec_linfo, li)
912 else
913 fname = string(li.func)
914 end
915 filename = short_path(li.file, filenamemap)
916 if showpointer
917 fname = string(
918 "0x",
919 string(li.pointer, base = 16, pad = 2*sizeof(Ptr{Cvoid})),
920 " ",
921 fname)
922 end
923 strs[i] = string(stroverhead, "", base, strcount, " ",
924 rtruncto(filename, widthfile),
925 ":",
926 li.line == -1 ? "?" : string(li.line),
927 "; ",
928 ltruncto(fname, widthfunc))
929 end
930 else
931 strs[i] = string(stroverhead, "", base, strcount, " [unknown stackframe]")
932 end
933 end
934 return strs
935 end

937 # turn a list of backtraces into a tree (implicitly separated by NULL markers)
938 function tree!(root::StackFrameTree{T}, all::Vector{UInt64}, lidict::Union{LineInfoFlatDict, LineInfoDict}, C::Bool, recur::Symbol,
939 threads::Union{Int,AbstractVector{Int},Nothing}=nothing, tasks::Union{UInt,AbstractVector{UInt},Nothing}=nothing) where {T}
940 !isempty(all) && !has_meta(all) && error("Profile data is missing required metadata")
941 parent = root
942 tops = Vector{StackFrameTree{T}}()
943 build = Vector{StackFrameTree{T}}()
944 startframe = length(all)
945 skip = false
946 nsleeping = 0
947 for i in startframe:-1:1
948 (startframe - 1) >= i >= (startframe - (nmeta + 1)) && continue # skip metadata (it's read ahead below) and extra block end NULL IP
949 ip = all[i]
950 if is_block_end(all, i)
951 # read metadata
952 thread_sleeping = all[i - META_OFFSET_SLEEPSTATE] - 1 # subtract 1 as state is incremented to avoid being equal to 0
953 # cpu_cycle_clock = all[i - META_OFFSET_CPUCYCLECLOCK]
954 taskid = all[i - META_OFFSET_TASKID]
955 threadid = all[i - META_OFFSET_THREADID]
956 if (threads !== nothing && !in(threadid, threads)) ||
957 (tasks !== nothing && !in(taskid, tasks))
958 skip = true
959 continue
960 end
961 if thread_sleeping == 1
962 nsleeping += 1
963 end
964 skip = false
965 # sentinel value indicates the start of a new backtrace
966 empty!(build)
967 root.recur = 0
968 if recur !== :off
969 # We mark all visited nodes to so we'll only count those branches
970 # once for each backtrace. Reset that now for the next backtrace.
971 push!(tops, parent)
972 for top in tops
973 while top.recur != 0
974 top.max_recur < top.recur && (top.max_recur = top.recur)
975 top.recur = 0
976 top = top.up
977 end
978 end
979 empty!(tops)
980 end
981 let this = parent
982 while this !== root
983 this.flat_count += 1
984 this = this.up
985 end
986 end
987 parent.overhead += 1
988 parent = root
989 root.count += 1
990 startframe = i
991 elseif !skip
992 if recur === :flat || recur === :flatc
993 pushfirst!(build, parent)
994 # Rewind the `parent` tree back, if this exact ip was already present *higher* in the current tree
995 found = false
996 for j in 1:(startframe - i)
997 if ip == all[i + j]
998 if recur === :flat # if not flattening C frames, check that now
999 frames = lidict[ip]
1000 frame = (frames isa Vector ? frames[1] : frames)
1001 frame.from_c && break # not flattening this frame
1002 end
1003 push!(tops, parent)
1004 parent = build[j]
1005 parent.recur += 1
1006 parent.count_recur += 1
1007 found = true
1008 break
1009 end
1010 end
1011 found && continue
1012 end
1013 builder_key = parent.builder_key
1014 builder_value = parent.builder_value
1015 fastkey = searchsortedfirst(builder_key, ip)
1016 if fastkey < length(builder_key) && builder_key[fastkey] === ip
1017 # jump forward to the end of the inlining chain
1018 # avoiding an extra (slow) lookup of `ip` in `lidict`
1019 # and an extra chain of them in `down`
1020 # note that we may even have this === parent (if we're ignoring this frame ip)
1021 this = builder_value[fastkey]
1022 let this = this
1023 while this !== parent && (recur === :off || this.recur == 0)
1024 this.count += 1
1025 this.recur = 1
1026 this = this.up
1027 end
1028 end
1029 parent = this
1030 continue
1031 end
1032
1033 frames = lidict[ip]
1034 nframes = (frames isa Vector ? length(frames) : 1)
1035 this = parent
1036 # add all the inlining frames
1037 for i = nframes:-1:1
1038 frame = (frames isa Vector ? frames[i] : frames)
1039 !C && frame.from_c && continue
1040 key = (T === UInt64 ? ip : frame)
1041 this = get!(StackFrameTree{T}, parent.down, key)
1042 if recur === :off || this.recur == 0
1043 this.frame = frame
1044 this.up = parent
1045 this.count += 1
1046 this.recur = 1
1047 end
1048 parent = this
1049 end
1050 # record where the end of this chain is for this ip
1051 insert!(builder_key, fastkey, ip)
1052 insert!(builder_value, fastkey, this)
1053 end
1054 end
1055 function cleanup!(node::StackFrameTree)
1056 stack = [node]
1057 while !isempty(stack)
1058 node = pop!(stack)
1059 node.recur = 0
1060 empty!(node.builder_key)
1061 empty!(node.builder_value)
1062 append!(stack, values(node.down))
1063 end
1064 nothing
1065 end
1066 cleanup!(root)
1067 return root, nsleeping
1068 end

1070 function maxstats(root::StackFrameTree)
1071 maxcount = Ref(0)
1072 maxflatcount = Ref(0)
1073 maxoverhead = Ref(0)
1074 maxmaxrecur = Ref(0)
1075 stack = [root]
1076 while !isempty(stack)
1077 node = pop!(stack)
1078 maxcount[] = max(maxcount[], node.count)
1079 maxoverhead[] = max(maxoverhead[], node.overhead)
1080 maxflatcount[] = max(maxflatcount[], node.flat_count)
1081 maxmaxrecur[] = max(maxmaxrecur[], node.max_recur)
1082 append!(stack, values(node.down))
1083 end
1084 return (count=maxcount[], count_flat=maxflatcount[], overhead=maxoverhead[], max_recur=maxmaxrecur[])
1085 end

1087 # Print the stack frame tree starting at a particular root. Uses a worklist to
1088 # avoid stack overflows.
1089 function print_tree(io::IO, bt::StackFrameTree{T}, cols::Int, fmt::ProfileFormat, is_subsection::Bool) where T
1090 maxes = maxstats(bt)
1091 filenamemap = Dict{Symbol,String}()
1092 worklist = [(bt, 0, 0, "")]
1093 if !is_subsection
1094 println(io, "Overhead [+additional indent] Count File:Line; Function")
1095 println(io, "=========================================================")
1096 end
1097 while !isempty(worklist)
1098 (bt, level, noisefloor, str) = popfirst!(worklist)
1099 isempty(str) || println(io, str)
1100 level > fmt.maxdepth && continue
1101 isempty(bt.down) && continue
1102 # Order the line information
1103 nexts = collect(values(bt.down))
1104 # Generate the string for each line
1105 strs = tree_format(nexts, level, cols, maxes, filenamemap, T === UInt64)
1106 # Recurse to the next level
1107 if fmt.sortedby === :count
1108 counts = collect(frame.count for frame in nexts)
1109 p = sortperm(counts)
1110 elseif fmt.sortedby === :overhead
1111 m = collect(frame.overhead for frame in nexts)
1112 p = sortperm(m)
1113 elseif fmt.sortedby === :flat_count
1114 m = collect(frame.flat_count for frame in nexts)
1115 p = sortperm(m)
1116 else
1117 lilist = collect(frame.frame for frame in nexts)
1118 p = liperm(lilist)
1119 end
1120 for i in reverse(p)
1121 down = nexts[i]
1122 count = down.count
1123 count < fmt.mincount && continue
1124 count < noisefloor && continue
1125 str = strs[i]
1126 noisefloor_down = fmt.noisefloor > 0 ? floor(Int, fmt.noisefloor * sqrt(count)) : 0
1127 pushfirst!(worklist, (down, level + 1, noisefloor_down, str))
1128 end
1129 end
1130 return
1131 end

1133 function tree(io::IO, data::Vector{UInt64}, lidict::Union{LineInfoFlatDict, LineInfoDict}, cols::Int, fmt::ProfileFormat,
1134 threads::Union{Int,AbstractVector{Int}}, tasks::Union{UInt,AbstractVector{UInt}}, is_subsection::Bool)
1135 if fmt.combine
1136 root, nsleeping = tree!(StackFrameTree{StackFrame}(), data, lidict, fmt.C, fmt.recur, threads, tasks)
1137 else
1138 root, nsleeping = tree!(StackFrameTree{UInt64}(), data, lidict, fmt.C, fmt.recur, threads, tasks)
1139 end
1140 util_perc = (1 - (nsleeping / root.count)) * 100
1141 !is_subsection && print_tree(io, root, cols, fmt, is_subsection)
1142 if isempty(root.down)
1143 if is_subsection
1144 Base.print(io, "Total snapshots: ")
1145 printstyled(io, "$(root.count)", color=Base.warn_color())
1146 Base.println(io, ". Utilization: ", round(Int, util_perc), "%")
1147 else
1148 warning_empty()
1149 end
1150 return true
1151 else
1152 Base.print(io, "Total snapshots: ", root.count, ". Utilization: ", round(Int, util_perc), "%")
1153 end
1154 if is_subsection
1155 println(io)
1156 print_tree(io, root, cols, fmt, is_subsection)
1157 else
1158 println(io, " across all threads and tasks. Use the `groupby` kwarg to break down by thread and/or task")
1159 end
1160 return false
1161 end

1163 function callersf(matchfunc::Function, bt::Vector, lidict::LineInfoFlatDict)
1164 counts = Dict{StackFrame, Int}()
1165 lastmatched = false
1166 for id in bt
1167 if id == 0
1168 lastmatched = false
1169 continue
1170 end
1171 li = lidict[id]
1172 if lastmatched
1173 if haskey(counts, li)
1174 counts[li] += 1
1175 else
1176 counts[li] = 1
1177 end
1178 end

1180 end

1182 v = collect(values(counts))
1183 p = sortperm(v, rev=true)

1206 # Order alphabetically (file, function) and then by line number
1207 function liperm(lilist::Vector{StackFrame})
1208 function lt(a::StackFrame, b::StackFrame)
1209 a == UNKNOWN && return false
1210 b == UNKNOWN && return true
1211 fcmp = cmp(a.file, b.file)
1212 fcmp < 0 && return true
1213 fcmp > 0 && return false
1214 fcmp = cmp(a.func, b.func)
1215 fcmp < 0 && return true
1216 fcmp > 0 && return false
1217 fcmp = cmp(a.line, b.line)
1218 fcmp < 0 && return true
1219 return false
1220 end
1221 return sortperm(lilist, lt = lt)
