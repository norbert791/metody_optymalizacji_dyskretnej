245 function compute_trycatch(code::Vector{Any}, ip::BitSet)
246 # The goal initially is to record the frame like this for the state at exit:
247 # 1: (enter 3) # == 0
248 # 3: (expr) # == 1
249 # 3: (leave 1) # == 1
250 # 4: (expr) # == 0
251 # then we can find all trys by walking backwards from :enter statements,
252 # and all catches by looking at the statement after the :enter
253 n = length(code)
254 empty!(ip)
255 ip.offset = 0 # for _bits_findnext
256 push!(ip, n + 1)
257 handler_at = fill(0, n)
258
259 # start from all :enter statements and record the location of the try
260 for pc = 1:n
261 stmt = code[pc]
262 if isexpr(stmt, :enter)
263 l = stmt.args[1]::Int
264 handler_at[pc + 1] = pc
265 push!(ip, pc + 1)
266 handler_at[l] = pc
267 push!(ip, l)
268 end
269 end
270
271 # now forward those marks to all :leave statements
272 pc´´ = 0
273 while true
274 # make progress on the active ip set
275 pc = _bits_findnext(ip.bits, pc´´)::Int
276 pc > n && break
277 while true # inner loop optimizes the common case where it can run straight from pc to pc + 1
278 pc´ = pc + 1 # next program-counter (after executing instruction)
279 if pc == pc´´
280 pc´´ = pc´
281 end
282 delete!(ip, pc)
283 cur_hand = handler_at[pc]
284 @assert cur_hand != 0 "unbalanced try/catch"
285 stmt = code[pc]
286 if isa(stmt, GotoNode)
287 pc´ = stmt.label
288 elseif isa(stmt, GotoIfNot)
289 l = stmt.dest::Int
290 if handler_at[l] != cur_hand
291 @assert handler_at[l] == 0 "unbalanced try/catch"
292 handler_at[l] = cur_hand
293 if l < pc´´
294 pc´´ = l
295 end
296 push!(ip, l)
297 end
298 elseif isa(stmt, ReturnNode)
299 @assert !isdefined(stmt, :val) "unbalanced try/catch"
300 break
301 elseif isa(stmt, Expr)
302 head = stmt.head
303 if head === :enter
304 cur_hand = pc
305 elseif head === :leave
306 l = stmt.args[1]::Int
307 for i = 1:l
308 cur_hand = handler_at[cur_hand]
309 end
310 cur_hand == 0 && break
311 end
312 end
313
314 pc´ > n && break # can't proceed with the fast-path fall-through
315 if handler_at[pc´] != cur_hand
316 @assert handler_at[pc´] == 0 "unbalanced try/catch"
317 handler_at[pc´] = cur_hand
318 elseif !in(pc´, ip)
319 break # already visited
320 end
321 pc = pc´
322 end
323 end
324
325 @assert first(ip) == n + 1
326 return handler_at
327 end

360 """
361 constrains_param(var::TypeVar, sig, covariant::Bool, type_constrains::Bool)
362
363 Check if `var` will be constrained to have a definite value
364 in any concrete leaftype subtype of `sig`.
365
366 It is used as a helper to determine whether type intersection is guaranteed to be able to
367 find a value for a particular type parameter.
368 A necessary condition for type intersection to not assign a parameter is that it only
369 appears in a `Union[All]` and during subtyping some other union component (that does not
370 constrain the type parameter) is selected.
371
372 The `type_constrains` flag determines whether Type{T} is considered to be constraining
373 `T`. This is not true in general, because of the existence of types with free type
374 parameters, however, some callers would like to ignore this corner case.
375 """
376 function constrains_param(var::TypeVar, @nospecialize(typ), covariant::Bool, type_constrains::Bool=false)
377 typ === var && return true
378 while typ isa UnionAll
379 covariant && constrains_param(var, typ.var.ub, covariant, type_constrains) && return true
380 # typ.var.lb doesn't constrain var
381 typ = typ.body
382 end
383 if typ isa Union
384 # for unions, verify that both options would constrain var
385 ba = constrains_param(var, typ.a, covariant, type_constrains)
386 bb = constrains_param(var, typ.b, covariant, type_constrains)
387 (ba && bb) && return true
388 elseif typ isa DataType
389 # return true if any param constrains var
390 fc = length(typ.parameters)
391 if fc > 0
392 if typ.name === Tuple.name
393 # vararg tuple needs special handling
394 for i in 1:(fc - 1)
395 p = typ.parameters[i]
396 constrains_param(var, p, covariant, type_constrains) && return true
397 end
398 lastp = typ.parameters[fc]
399 vararg = unwrap_unionall(lastp)
400 if vararg isa Core.TypeofVararg && isdefined(vararg, :N)
401 constrains_param(var, vararg.N, covariant, type_constrains) && return true
402 # T = vararg.parameters[1] doesn't constrain var
403 else
404 constrains_param(var, lastp, covariant, type_constrains) && return true
405 end
406 else
407 if typ.name === typename(Type) && typ.parameters[1] === var && var.ub === Any
408 # Types with free type parameters are <: Type cause the typevar
409 # to be unconstrained because Type{T} with free typevars is illegal
410 return type_constrains
411 end
412 for i in 1:fc
413 p = typ.parameters[i]
414 constrains_param(var, p, false, type_constrains) && return true
415 end
416 end
417 end
418 end
419 return false
420 end

537 function record_ssa_assign!(𝕃ᵢ::AbstractLattice, ssa_id::Int, @nospecialize(new), frame::InferenceState)
538 ssavaluetypes = frame.ssavaluetypes
539 old = ssavaluetypes[ssa_id]
540 if old === NOT_FOUND || !⊑(𝕃ᵢ, new, old)
541 # typically, we expect that old ⊑ new (that output information only
542 # gets less precise with worse input information), but to actually
543 # guarantee convergence we need to use tmerge here to ensure that is true
544 ssavaluetypes[ssa_id] = old === NOT_FOUND ? new : tmerge(𝕃ᵢ, old, new)
545 W = frame.ip
546 for r in frame.ssavalue_uses[ssa_id]
547 if was_reached(frame, r)
548 usebb = block_for_inst(frame.cfg, r)
549 # We're guaranteed to visit the statement if it's in the current
550 # basic block, since SSA values can only ever appear after their
551 # def.
552 if usebb != frame.currbb
553 push!(W, usebb)
554 end
555 end
556 end
557 end
558 return nothing
559 end
